<script>

  const API_URL = "https://script.google.com/macros/s/AKfycbz9IwmCMlitOguVwfjPWpcVWEDBu-hKy-5v_kzijsklsWAM4JE1AmdPhsMp2oQIK6WJ/exec?key=notewebapp”;


  async function loadData() {

    const res = await fetch(API_URL);

    const data = await res.json();


    const list = document.getElementById("list");

    list.innerHTML = "";


    data.forEach(row => {

      const div = document.createElement("div");

      div.className = "item";

      div.innerHTML = `

        <span>${row.項目}</span>

        <strong>${row.餘額}</strong>

      `;

      list.appendChild(div);

    });

  }


  // 頁面一開就載入

  loadData();

</script>
